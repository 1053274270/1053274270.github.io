void Model::setStepTab(ResultStep *step)
{
    int stepTabState = stepTabExisted(step);
    qDebug()<<"stepTabState:"<<stepTabState;
    if(stepTabState == -1){
        mystepResult[currentStep]->openOpenGL();
        CentralTabWidget->addTab(step->myOpenGL, step->name);
        int num_tab = CentralTabWidget->count()-1;
        CentralTabWidget->setCurrentIndex(num_tab);
    }else if(CentralTabWidget->tabText(CentralTabWidget->currentIndex())!=
             mystepResult[currentStep]->name){
        CentralTabWidget->setCurrentIndex(stepTabState);
    }else {
        showResult();  //根据当前载荷步currentStep和mystepResult[currentStep]->showType显示结果
    }
}

void Model::showResult()
{
    switch (mystepResult[currentStep]->showType) {
    case 1:
        showStress();
        break;
    case 2:
        showStrain();
        break;
    case 3:
        showDeformation();
        break;
    default:
        break;
    }
    return;
}

void Model::openResult( QProgressBar *pProgressBar)
{
    if(0<=currentStep && currentStep<mystepResult.size()){
        mystepResult[currentStep]->openResult( pProgressBar);
        if(mystepResult[currentStep]->deformation!=nullptr)
            myresultPanal->setModelRange(
                    mystepResult[currentStep]->deformation->modelRange);
        ShowResult = true;
        ResetView();
    }
    return;
}

void ResultStep::openResult( QProgressBar *pProgressBar)
{
    QVector<STRU_Member> *element;
    QVector<gNode> *node;
    QVector<co> *deform;

    switch (showType) {
    case 1:
        if(deformation!=nullptr)
            deformation->openDeformation(pProgressBar);
        element = &(deformation->element);
        node = &(deformation->allNode.node);
        deform = &(deformation->deformation);
        if(stress!=nullptr)
            stress->openResult(pProgressBar, element, node,
                               deform);
        break;
    case 2:
        if(deformation!=nullptr)
            deformation->openDeformation(pProgressBar);
        element = &(deformation->element);
        node = &(deformation->allNode.node);
        deform = &(deformation->deformation);
        if(strain!=nullptr)
            strain->openResult(pProgressBar, element, node,
                               deform);
        break;
    case 3:
        if(deformation!=nullptr)
            deformation->openDeformation(pProgressBar);
        break;
    default:
        break;
    }
    if(myOpenGL==nullptr) return;
    myOpenGL->modelRange.enlarge(deformation->modelRange,0.0,0,0,0);
    /*myOpenGL->modelRange = deformation->modelRange;
    myOpenGL->modelRange.x.min = deformation->modelRange.x.min-
            (deformation->modelRange.x.max-deformation->modelRange.x.min)*0.2;
    myOpenGL->modelRange.y.min = deformation->modelRange.y.min;*/
    myOpenGL->update();
    return;
}
